# 缓存架构设计

## 概述

多级缓存架构是提升系统性能的关键手段。本规范采用 L1（本地内存）→ L2（Redis）→ L3（数据库）的三级缓存策略，涵盖缓存策略选择、TTL 设计、异常防护以及 Python 实现方法。

## 核心规则

### 🔴 必须遵守 (MUST)

1. **实现多级缓存架构**
   - L1：本地内存缓存（函数级，毫秒级响应）
   - L2：分布式缓存（Redis，亚毫秒级响应）
   - L3：数据库（最终持久化）
   - 查询顺序：L1 → L2 → L3，逐级回退

2. **缓存 Key 设计必须规范**
   - 格式：`{prefix}:{module}:{entity}:{identifier}`
   - 示例：`cache:search:documents:fastapi_router`
   - Key 必须包含版本号或具备自动失效能力

3. **必须防护缓存异常**
   - 缓存穿透：查询不存在的数据
   - 缓存击穿：热点 Key 失效
   - 缓存雪崩：大量 Key 同时失效

### 🟡 强烈建议 (SHOULD)

1. **Cache-Aside 作为默认策略**
   - 读取：先查缓存，未命中则查数据库并回写缓存
   - 写入：先更新数据库，再删除缓存

2. **TTL 差异化设计**
   - 热点数据：长 TTL（5-30 分钟），配合主动刷新
   - 实时数据：短 TTL（10-60 秒）
   - 静态数据：超长 TTL（1-24 小时）

3. **缓存预热**
   - 服务启动时加载热点数据到缓存
   - 避免服务启动后的大量缓存未命中

### 🟢 可选建议 (MAY)

1. **Read-Through / Write-Behind**
   - Read-Through：缓存层自动回源数据库
   - Write-Behind：先写缓存，异步写数据库

2. **本地缓存二级失效**
   - 通过 Redis Pub/Sub 广播本地缓存失效消息

## 正确示例

```python
import asyncio
import json
import hashlib
from typing import Any, Callable, Optional
from functools import lru_cache
from dataclasses import dataclass
import time


# ========== L1: 本地内存缓存 ==========

class LRUCache:
    """LRU 本地缓存（线程安全）"""
    
    def __init__(self, capacity: int = 1000, ttl: int = 300):
        self.capacity = capacity
        self.ttl = ttl
        self._cache: dict[str, tuple[Any, float]] = {}
        self._access_order: list[str] = []
    
    def get(self, key: str) -> Any | None:
        if key not in self._cache:
            return None
        
        value, expire_time = self._cache[key]
        if time.time() > expire_time:
            self._remove(key)
            return None
        
        # 更新访问顺序
        self._access_order.remove(key)
        self._access_order.append(key)
        return value
    
    def set(self, key: str, value: Any, ttl: int | None = None):
        if ttl is None:
            ttl = self.ttl
        
        if len(self._cache) >= self.capacity:
            oldest = self._access_order.pop(0)
            self._cache.pop(oldest, None)
        
        self._cache[key] = (value, time.time() + ttl)
        if key in self._access_order:
            self._access_order.remove(key)
        self._access_order.append(key)
    
    def _remove(self, key: str):
        self._cache.pop(key, None)
        if key in self._access_order:
            self._access_order.remove(key)
    
    def clear(self):
        self._cache.clear()
        self._access_order.clear()


# ========== L2: Redis 缓存 ==========

class RedisCache:
    """Redis 分布式缓存"""
    
    def __init__(self, redis_client):
        self.redis = redis_client
        self.default_ttl = 300  # 5 分钟
    
    async def get(self, key: str) -> Any | None:
        data = await self.redis.get(key)
        if data is None:
            return None
        try:
            return json.loads(data)
        except json.JSONDecodeError:
            return data
    
    async def set(self, key: str, value: Any, ttl: int | None = None):
        ttl = ttl or self.default_ttl
        if isinstance(value, (dict, list)):
            value = json.dumps(value, ensure_ascii=False)
        await self.redis.setex(key, ttl, value)
    
    async def delete(self, key: str):
        await self.redis.delete(key)
    
    async def exists(self, key: str) -> bool:
        return await self.redis.exists(key) > 0


# ========== 多级缓存管理器 ==========

@dataclass
class CacheConfig:
    prefix: str = "cache"
    l1_capacity: int = 1000
    l1_ttl: int = 60        # L1 短 TTL
    l2_ttl: int = 300       # L2 中 TTL
    enable_l1: bool = True
    enable_l2: bool = True


class MultiLevelCache:
    """三级缓存管理器：L1(LRU) → L2(Redis) → L3(DB)"""
    
    def __init__(self, config: CacheConfig, redis_client=None):
        self.config = config
        self.l1 = LRUCache(capacity=config.l1_capacity, ttl=config.l1_ttl)
        self.l2 = RedisCache(redis_client) if redis_client else None
        self._stats = {"l1_hits": 0, "l2_hits": 0, "l3_hits": 0, "misses": 0}
    
    def _make_key(self, module: str, entity: str, identifier: str) -> str:
        return f"{self.config.prefix}:{module}:{entity}:{identifier}"
    
    async def get(self, module: str, entity: str, identifier: str,
                  fetcher: Callable | None = None) -> Any | None:
        """
        多级缓存查询
        顺序: L1 → L2 → L3(fetcher)
        """
        cache_key = self._make_key(module, entity, identifier)
        
        # L1: 本地缓存
        if self.config.enable_l1:
            result = self.l1.get(cache_key)
            if result is not None:
                self._stats["l1_hits"] += 1
                return result
        
        # L2: Redis 缓存
        if self.config.enable_l2 and self.l2:
            result = await self.l2.get(cache_key)
            if result is not None:
                self._stats["l2_hits"] += 1
                # 回写 L1
                if self.config.enable_l1:
                    self.l1.set(cache_key, result)
                return result
        
        # L3: 数据库回源
        if fetcher:
            result = await fetcher()
            if result is not None:
                self._stats["l3_hits"] += 1
                # 回写 L2 和 L1
                if self.config.enable_l2 and self.l2:
                    await self.l2.set(cache_key, result)
                if self.config.enable_l1:
                    self.l1.set(cache_key, result)
                return result
        
        self._stats["misses"] += 1
        return None
    
    async def invalidate(self, module: str, entity: str, identifier: str):
        """使缓存失效"""
        cache_key = self._make_key(module, entity, identifier)
        
        # 删除 L1
        if self.config.enable_l1:
            self.l1._remove(cache_key)
        
        # 删除 L2
        if self.config.enable_l2 and self.l2:
            await self.l2.delete(cache_key)
    
    def get_stats(self) -> dict:
        total = sum(self._stats.values())
        return {
            **self._stats,
            "hit_rate": (self._stats["l1_hits"] + self._stats["l2_hits"] + self._stats["l3_hits"]) / total if total > 0 else 0,
        }


# ========== 缓存异常防护 ==========

class CacheProtection:
    """缓存穿透/击穿/雪崩 防护"""
    
    @staticmethod
    def bloom_filter_prevention():
        """
        缓存穿透防护：布隆过滤器
        查询前先检查布隆过滤器，不存在的数据直接返回
        """
        from pybloom_live import BloomFilter
        
        bloom = BloomFilter(capacity=1000000, error_rate=0.001)
        
        def check_exists(key: str) -> bool:
            if key not in bloom:
                return False  # 一定不存在
            return True  # 可能存在
        
        return check_exists
    
    @staticmethod
    def mutex_prevention():
        """
        缓存击穿防护：互斥锁
        热点 Key 失效时，只允许一个请求回源数据库
        """
        mutex_lock = asyncio.Lock()
        
        async def get_hot_key(cache, key: str, fetcher: Callable):
            # 先尝试获取缓存
            result = await cache.get(key)
            if result is not None:
                return result
            
            # 缓存未命中，加锁回源
            async with mutex_lock:
                # 双重检查
                result = await cache.get(key)
                if result is not None:
                    return result
                
                result = await fetcher()
                await cache.set(key, result)
                return result
        
        return get_hot_key
```

## 错误示例

```python
# 错误 1：缓存 Key 设计不当

def bad_cache_key():
    """Key 没有区分度或过长"""
    
    # Key 过短，碰撞率高
    key = f"user:{id}"  # 不同查询的 Key 可能相同
    
    # Key 过长，浪费 Redis 内存
    key = f"cache:{module}:{entity}:{identifier}:{timestamp}:{random_id}:{user_id}"
    # 163 字节的 Key，存储 100 万条 → 163MB 元数据
    # 应控制 Key 长度在 50 字节以内
    
    # Key 不含版本，导致旧数据无法失效
    key = "user_profile"  # 业务变更后 Key 不变，造成数据错乱


# 错误 2：没有缓存异常防护

def bad_no_protection():
    """不做防护 → 高并发时系统崩溃"""
    
    # 缓存穿透：查询 ID=-1 的记录
    # 缓存中没有，每次都查数据库
    # 恶意攻击利用此漏洞打垮数据库
    
    # 缓存击穿：热搜新闻 Key 失效
    # 10 万请求同时打到数据库
    # 数据库连接池耗尽
    
    # 缓存雪崩：所有 Key 同一时间失效
    # 例如 TTL 都设为 3600 秒
    # 整点时刻全部失效


# 错误 3：缓存与数据库不一致

def bad_inconsistent():
    """更新数据库后没有处理缓存"""
    
    def update_user(user_id: str, data: dict):
        # 只更新数据库
        db.update_user(user_id, data)
        # 没有删除或更新缓存
        # 下次读取仍是旧数据


# 错误 4：所有数据使用相同 TTL

def bad_same_ttl():
    """所有缓存 Key 都用同样的 TTL"""
    
    ttl = 300  # 5 分钟
    # 用户配置（几小时不变）→ TTL 太短，浪费
    # 实时库存（秒级变化）→ TTL 太长，数据过期
    # 应差异化设置


# 错误 5：过度缓存

def bad_over_caching():
    """把不需要缓存的数据也缓存了"""
    
    # 缓存了 10 万条用户信息
    # 但只有 100 个活跃用户
    # 99.9% 的缓存空间浪费
```
