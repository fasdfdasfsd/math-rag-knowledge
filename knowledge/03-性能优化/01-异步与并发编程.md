# 异步与并发编程

## 概述

Python 异步编程是构建高性能 I/O 密集型应用的基础。本规范涵盖 asyncio 最佳实践、线程池与协程的选择、FastAPI 异步路由以及数据库异步驱动的使用，确保应用在高并发场景下的稳定性和效率。

## 核心规则

### 🔴 必须遵守 (MUST)

1. **异步函数中不得调用同步 I/O 阻塞事件循环**
   - 禁止在 `async def` 中直接调用 `time.sleep()`
   - 禁止在 `async def` 中直接调用同步文件 I/O、requests、数据库驱动
   - 必须使用 `asyncio.sleep()`、`aiofiles`、`httpx.AsyncClient`、`asyncpg` 等异步替代

2. **I/O 密集型任务必须使用 asyncio**
   - 网络请求、文件读写、数据库操作等 I/O 操作优先使用异步
   - CPU 密集型任务使用 `asyncio.to_thread()` 或进程池

3. **API 并发必须使用 Semaphore 控制**
   - 对外部 API 的并发请求数必须有上限
   - 防止因并发过高被限流或导致服务宕机

### 🟡 强烈建议 (SHOULD)

1. **使用 TaskGroup 管理并发任务**
   - Python 3.11+ 推荐使用 `asyncio.TaskGroup` 替代 `asyncio.gather`
   - TaskGroup 提供更好的错误处理和任务管理

2. **FastAPI 路由统一使用 async def**
   - 即使内部没有 await，也使用 async def 保持一致性
   - 同步函数 FastAPI 会自动在线程池中执行，增加上下文切换开销

3. **数据库使用异步驱动**
   - PostgreSQL：`asyncpg`
   - Redis：`redis.asyncio`
   - MongoDB：`motor`

### 🟢 可选建议 (MAY)

1. **uvloop 替换默认事件循环**
   - uvloop 基于 libuv，性能接近 Node.js
   - 通过 `asyncio.set_event_loop_policy(uvloop.EventLoopPolicy())` 启用

2. **结构化并发**
   - 使用 `asyncio.timeout()` 设置超时
   - 使用 `asyncio.CancelledError` 处理取消信号

## 正确示例

```python
import asyncio
import time
from typing import List, Any
from dataclasses import dataclass


# ========== 基础异步模式 ==========

async def async_sleep_example():
    """正确：使用 asyncio.sleep 而非 time.sleep"""
    print("开始等待...")
    await asyncio.sleep(1)  # 正确：非阻塞
    print("等待结束")


async def async_http_example():
    """正确：使用异步 HTTP 客户端"""
    import httpx
    
    async with httpx.AsyncClient() as client:
        response = await client.get("https://api.example.com/data")
        return response.json()


async def async_file_example():
    """正确：使用异步文件 I/O"""
    import aiofiles
    
    async with aiofiles.open("data.txt", mode="r") as f:
        content = await f.read()
    return content


# ========== Semaphore 并发控制 ==========

class APIClient:
    """带并发控制的 API 客户端"""
    
    def __init__(self, max_concurrent: int = 5):
        self.semaphore = asyncio.Semaphore(max_concurrent)
        self.client = httpx.AsyncClient(timeout=30.0)
    
    async def call_api(self, endpoint: str) -> dict:
        """带并发限制的 API 调用"""
        async with self.semaphore:  # 并发控制
            print(f"[API] 调用 {endpoint} (当前并发: {self._current_concurrency()})")
            response = await self.client.get(endpoint)
            response.raise_for_status()
            return response.json()
    
    def _current_concurrency(self) -> int:
        """估算当前并发数"""
        return self.semaphore._value  # 仅供调试


# ========== TaskGroup 并发管理 ==========

async def process_batch(items: List[str]) -> List[dict]:
    """使用 TaskGroup 并发处理批量任务"""
    client = APIClient(max_concurrent=5)
    
    async with asyncio.TaskGroup() as tg:
        tasks = [tg.create_task(client.call_api(item)) for item in items]
        # TaskGroup 自动管理所有任务的生命周期
    
    results = [t.result() for t in tasks]
    return results


# ========== CPU 密集任务处理 ==========

def cpu_intensive_task(data: List[int]) -> int:
    """CPU 密集型任务（同步函数）"""
    total = 0
    for i in data:
        total += i * i
        # 模拟复杂计算
        for _ in range(1000):
            total += 1
    return total


async def handle_cpu_task(data: List[int]) -> int:
    """正确：将 CPU 密集任务放到线程池执行"""
    result = await asyncio.to_thread(cpu_intensive_task, data)
    return result


# ========== FastAPI 异步路由 ==========

from fastapi import FastAPI, HTTPException, Depends
from pydantic import BaseModel

app = FastAPI()


class QueryRequest(BaseModel):
    query: str
    max_results: int = 5


@app.get("/health")
async def health_check():
    """健康检查 — 使用 async def"""
    return {"status": "ok"}


@app.post("/search")
async def search(request: QueryRequest):
    """搜索接口 — 全异步实现"""
    try:
        # 异步数据库查询
        results = await search_database(request.query, request.max_results)
        return {"results": results}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))


# ========== 数据库异步操作 ==========

import asyncpg

class AsyncDatabase:
    """异步数据库连接池"""
    
    def __init__(self, dsn: str, min_connections: int = 5, max_connections: int = 20):
        self.dsn = dsn
        self.min_connections = min_connections
        self.max_connections = max_connections
        self.pool: asyncpg.Pool | None = None
    
    async def connect(self):
        self.pool = await asyncpg.create_pool(
            self.dsn,
            min_size=self.min_connections,
            max_size=self.max_connections,
        )
    
    async def search_documents(self, query: str, limit: int = 5) -> List[dict]:
        """异步向量搜索（假设有 pgvector）"""
        async with self.pool.acquire() as conn:
            rows = await conn.fetch(
                """
                SELECT id, content, metadata,
                       1 - (embedding <=> $1) AS similarity
                FROM documents
                WHERE content ILIKE $2
                ORDER BY similarity DESC
                LIMIT $3
                """,
                query, f"%{query}%", limit
            )
            return [dict(row) for row in rows]
```

## 错误示例

```python
# 错误 1：async 函数中调用同步 I/O 阻塞事件循环

async def bad_blocking_io():
    """错误：time.sleep 阻塞整个事件循环"""
    print("开始等待...")
    time.sleep(5)  # 同步阻塞！整个事件循环暂停 5 秒
    # 期间所有其他协程都无法执行
    print("等待结束")


async def bad_sync_http():
    """错误：同步 requests 阻塞事件循环"""
    import requests
    response = requests.get("https://api.example.com/data")  # 阻塞！
    # 替代方案：httpx.AsyncClient
    return response.json()


# 错误 2：无限制并发

async def bad_unlimited_concurrency():
    """错误：没有 Semaphore 控制并发"""
    async with httpx.AsyncClient() as client:
        # 1000 个请求同时发出
        # → API 限流 / 连接池耗尽 / OOM
        tasks = [client.get(f"https://api.example.com/item/{i}") 
                 for i in range(1000)]
        responses = await asyncio.gather(*tasks)


# 错误 3：FastAPI 使用 def 而非 async def

"""
@app.get("/search")
def search(q: str):  # 错误：使用同步 def
    # FastAPI 会在线程池中运行
    # 实际上没有 await，但：
    # 1. 线程上下文切换开销
    # 2. 如果内部调用异步代码需要额外处理
    return {"result": "data"}
"""


# 错误 4：异步函数中混用同步数据库驱动

async def bad_sync_db():
    """错误：在 async 函数中使用同步数据库驱动"""
    import psycopg2  # 同步驱动
    conn = psycopg2.connect("...")  # 阻塞！
    cur = conn.cursor()
    cur.execute("SELECT * FROM documents")  # 阻塞！
    # 应使用 asyncpg


# 错误 5：忘记 await 导致协程未执行

async def bad_missing_await():
    """错误：忘记 await"""
    async def fetch_data():
        await asyncio.sleep(1)
        return "data"
    
    result = fetch_data()  # 没有 await！
    # result 是 coroutine 对象，不是实际数据
    print(result)  # <coroutine object ...>
```

## 工具链配置

```python
# uvloop 事件循环配置
try:
    import uvloop
    asyncio.set_event_loop_policy(uvloop.EventLoopPolicy())
except ImportError:
    pass  # 没有 uvloop 时使用默认事件循环


# 超时控制
async def with_timeout(coro, timeout: float = 10.0):
    """为协程设置超时"""
    try:
        async with asyncio.timeout(timeout):
            return await coro
    except TimeoutError:
        print(f"[Timeout] 操作超时 ({timeout}s)")
        raise
```
