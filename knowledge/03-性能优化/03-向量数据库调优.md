# 向量数据库调优

## 概述

向量数据库是 RAG 系统的核心组件，其性能直接影响检索速度和成本。本规范以 pgvector 为例，涵盖索引选择、参数调优、量化技术和批量优化，帮助在不同场景下实现最佳的查询性能与存储效率。

## 核心规则

### 🔴 必须遵守 (MUST)

1. **数据量超过 10 万条时必须创建索引**
   - 无索引的向量搜索是 O(n) 全表扫描
   - 索引选择：IVFFlat 适合构建快，HNSW 适合查询快

2. **查询前设置正确的 probes/ef_search 参数**
   - IVFFlat：`SET ivfflat.probes = N`
   - HNSW：`SET hnsw.ef_search = N`
   - 默认 probes=1 召回率很低，需根据数据量调大

### 🟡 强烈建议 (SHOULD)

1. **索引参数调优**
   - IVFFlat：`lists` 建议为 `sqrt(rows)`，`probes` 从 10 开始调试
   - HNSW：`m`（每层最大连接数）默认 16，`ef_construction` 默认 200，`ef_search` 从 50 开始调试

2. **批量插入使用 COPY 而非逐条 INSERT**
   - COPY 比逐条 INSERT 快 5-10 倍
   - 批量大小建议 1000-5000 条/批

3. **存储优化考虑量化**
   - PQ（Product Quantization）：压缩 4-8 倍，精度损失 1-3%
   - SQ（Scalar Quantization）：将 float32 转为 int8，存储减 75%

### 🟢 可选建议 (MAY)

1. **并行查询**
   - 分区表 + 并行查询
   - 预过滤 + 向量搜索

2. **异步索引构建**
   - 数据入库后异步构建索引
   - 避免阻塞写入

## 正确示例

```python
from typing import List, Optional
import asyncpg
import time


class VectorDatabaseOptimizer:
    """向量数据库优化器"""
    
    def __init__(self, pool: asyncpg.Pool):
        self.pool = pool
    
    async def create_ivfflat_index(self, table: str = "documents",
                                    lists: int = 100):
        """创建 IVFFlat 索引"""
        async with self.pool.acquire() as conn:
            # 先删除旧索引
            await conn.execute(f"DROP INDEX IF EXISTS idx_{table}_embedding")
            
            # 创建 IVFFlat 索引
            await conn.execute(f"""
                CREATE INDEX idx_{table}_embedding
                ON {table} USING ivfflat (embedding vector_cosine_ops)
                WITH (lists = {lists});
            """)
            
            print(f"[VectorDB] 创建 IVFFlat 索引完成 (lists={lists})")
    
    async def create_hnsw_index(self, table: str = "documents",
                                 m: int = 16, ef_construction: int = 200):
        """创建 HNSW 索引"""
        async with self.pool.acquire() as conn:
            await conn.execute(f"DROP INDEX IF EXISTS idx_{table}_embedding_hnsw")
            
            await conn.execute(f"""
                CREATE INDEX idx_{table}_embedding_hnsw
                ON {table} USING hnsw (embedding vector_cosine_ops)
                WITH (m = {m}, ef_construction = {ef_construction});
            """)
            
            print(f"[VectorDB] 创建 HNSW 索引完成 (m={m}, ef_construction={ef_construction})")
    
    async def search_with_tuning(self, query_embedding: List[float],
                                  top_k: int = 10,
                                  probes: int = 10,
                                  ef_search: int = 50,
                                  use_hnsw: bool = False) -> List[dict]:
        """调优参数后的向量搜索"""
        async with self.pool.acquire() as conn:
            # 设置搜索参数
            if not use_hnsw:
                await conn.execute(f"SET ivfflat.probes = {probes}")
            else:
                await conn.execute(f"SET hnsw.ef_search = {ef_search}")
            
            # 执行搜索
            rows = await conn.fetch(f"""
                SELECT id, content, metadata,
                       1 - (embedding <=> $1) AS similarity
                FROM documents
                ORDER BY embedding <=> $1
                LIMIT $2
            """, query_embedding, top_k)
            
            return [dict(row) for row in rows]
    
    async def bulk_insert(self, records: List[tuple], batch_size: int = 1000):
        """批量插入优化：使用 COPY 而非 INSERT"""
        import io
        import csv
        
        total = len(records)
        print(f"[VectorDB] 开始批量插入 {total} 条记录")
        start = time.time()
        
        async with self.pool.acquire() as conn:
            for i in range(0, total, batch_size):
                batch = records[i:i + batch_size]
                
                # 使用 COPY 批量导入
                buffer = io.StringIO()
                writer = csv.writer(buffer)
                for record in batch:
                    writer.writerow(record)
                buffer.seek(0)
                
                await conn.copy_from(buffer, "documents", 
                                     columns=["id", "content", "embedding"],
                                     format="csv")
                
                elapsed = time.time() - start
                print(f"[VectorDB] 已插入 {min(i+batch_size, total)}/{total} "
                      f"({elapsed:.1f}s)")
        
        print(f"[VectorDB] 批量插入完成: {total} 条, {time.time()-start:.1f}s")
    
    async def optimize_query(self, query_text: str):
        """查询优化：预过滤 + 向量搜索"""
        async with self.pool.acquire() as conn:
            # 使用 CTE 先过滤元数据，再向量搜索
            rows = await conn.fetch(f"""
                WITH filtered AS (
                    SELECT id, content, embedding, metadata
                    FROM documents
                    WHERE metadata->>'status' = 'active'
                      AND metadata->>'language' = 'zh'
                )
                SELECT id, content, metadata,
                       1 - (embedding <=> $1) AS similarity
                FROM filtered
                ORDER BY similarity DESC
                LIMIT $2
            """, query_text, 10)
            
            return [dict(row) for row in rows]


# 索引选择决策器
class IndexSelector:
    """根据使用场景推荐索引类型"""
    
    @staticmethod
    def recommend(dataset_size: int, 
                  query_frequency: int, 
                  update_frequency: int,
                  latency_sensitive: bool) -> dict:
        """
        推荐索引类型
        Args:
            dataset_size: 数据集大小
            query_frequency: 查询频率（每秒）
            update_frequency: 更新频率（每天）
            latency_sensitive: 是否延迟敏感
        Returns:
            推荐配置
        """
        if dataset_size < 100_000:
            return {
                "index": "none",
                "reason": "数据量小，全表扫描已足够",
            }
        
        if update_frequency > 1000:
            # 高频更新场景，IVFFlat 构建快
            lists = int(dataset_size ** 0.5)
            return {
                "index": "ivfflat",
                "lists": min(lists, 4000),
                "probes": 10,
                "reason": "高频更新，IVFFlat 索引构建快",
            }
        
        if latency_sensitive or query_frequency > 100:
            # 低延迟/高查询场景，HNSW 查询快
            return {
                "index": "hnsw",
                "m": 16,
                "ef_construction": 200,
                "ef_search": 50,
                "reason": "查询密集场景，HNSW 查询延时长于 IVFFlat",
            }
        
        # 默认：IVFFlat
        lists = int(dataset_size ** 0.5)
        return {
            "index": "ivfflat",
            "lists": min(lists, 4000),
            "probes": 10,
            "reason": "通用场景，IVFFlat 平衡查询和构建性能",
        }
```

## 错误示例

```python
# 错误 1：没有创建索引

def bad_no_index():
    """10 万条数据无索引"""
    
    # SELECT * FROM documents ORDER BY embedding <=> $1 LIMIT 10;
    # → 每次查询都是全表扫描
    # → 10 万条时延时 > 1 秒
    # → 100 万条时延时 > 10 秒


# 错误 2：IVFFlat probes 参数不调优

def bad_default_probes():
    """使用 probes=1（默认值）"""
    
    # SET ivfflat.probes = 1;
    # 只搜索 1 个聚类，召回率可能 < 50%
    # 很多相关文档被遗漏
    # 应设置为 lists / 10 或至少 10-50


# 错误 3：逐条 INSERT

def bad_row_by_row_insert():
    """逐条插入 10 万条数据"""
    
    for record in records:
        conn.execute("INSERT INTO documents VALUES ($1, $2, $3)", record)
    # 10 万次事务提交
    # 比 COPY 慢 5-10 倍


# 错误 4：hnsw ef_search 设置过大

def bad_ef_search_too_large():
    """ef_search=1000 → 查询极慢"""
    
    # SET hnsw.ef_search = 1000;
    # ef_search 越大召回率越高，但查询越慢
    # ef_search=1000 可能比 ef_search=50 慢 20 倍
    # 但召回率提升有限（从 0.95 到 0.97）


# 错误 5：索引构建时机不当

def bad_index_build_timing():
    """每次插入都重建索引"""
    
    for record in records:
        conn.execute("INSERT INTO documents ...")
        conn.execute("REINDEX INDEX idx_documents_embedding")  # 错误！
    # 索引重建 O(n log n)，每次插入都重建成本极高
    # 应批量插入后再一次性构建
```

## 工具链配置

```sql
-- pgvector 初始化 SQL
CREATE EXTENSION IF NOT EXISTS vector;

-- 创建表
CREATE TABLE documents (
    id SERIAL PRIMARY KEY,
    content TEXT NOT NULL,
    embedding vector(1536),
    metadata JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT NOW()
);

-- 索引参数调优示例
-- 
-- 场景 1：100 万条数据，每天更新 1000 条
-- 推荐：IVFFlat
CREATE INDEX idx_documents_ivfflat
ON documents USING ivfflat (embedding vector_cosine_ops)
WITH (lists = 1000);  -- sqrt(1_000_000) = 1000

SET ivfflat.probes = 50;  -- 1000 / 20 = 50

-- 
-- 场景 2：100 万条数据，高并发查询
-- 推荐：HNSW
CREATE INDEX idx_documents_hnsw
ON documents USING hnsw (embedding vector_cosine_ops)
WITH (m = 16, ef_construction = 200);

SET hnsw.ef_search = 100;
```
