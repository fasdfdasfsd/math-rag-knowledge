# 大文件与流式处理

## 概述

处理大文件（GB 级别）时，全量加载到内存会导致 OOM 或性能严重下降。本规范涵盖使用 Generator/yield 流式处理、FastAPI StreamingResponse、分块上传/下载以及内存管理策略，确保系统能高效处理大规模数据。

## 核心规则

### 🔴 必须遵守 (MUST)

1. **大文件禁止全量加载到内存**
   - 使用生成器（Generator）分批读取和写入
   - 控制每批处理的数据量（建议 1-10MB/批）

2. **API 响应体超过 100MB 时必须使用流式响应**
   - 使用 FastAPI `StreamingResponse`
   - 使用异步生成器逐步产出数据

### 🟡 强烈建议 (SHOULD)

1. **分块上传/下载**
   - 上传：客户端分片上传，服务端逐个接收
   - 下载：支持 Range 头部，实现断点续传

2. **内存监控**
   - 使用 `tracemalloc` 或 `psutil` 监控内存使用
   - 设置进程级内存上限

3. **临时文件清理**
   - 使用 `tempfile` 模块管理临时文件
   - 确保 finally 或上下文管理器中清理

### 🟢 可选建议 (MAY)

1. **内存映射文件**
   - 使用 `mmap` 对大文件进行内存映射
   - 适合随机访问场景

2. **异步文件流**
   - 使用 `aiofiles` 实现异步大文件处理

## 正确示例

```python
import asyncio
import os
from typing import AsyncGenerator, List, Iterator
from fastapi import FastAPI, UploadFile, HTTPException
from fastapi.responses import StreamingResponse
import aiofiles
import json


# ========== 流式读取大文件 ==========

async def read_large_file_async(file_path: str, 
                                 chunk_size: int = 8192) -> AsyncGenerator[bytes, None]:
    """异步流式读取大文件"""
    async with aiofiles.open(file_path, "rb") as f:
        while True:
            chunk = await f.read(chunk_size)
            if not chunk:
                break
            yield chunk


async def process_large_file_generator(file_path: str) -> AsyncGenerator[str, None]:
    """使用生成器逐行处理大文件"""
    async with aiofiles.open(file_path, "r", encoding="utf-8") as f:
        async for line in f:
            # 逐行处理，每行独立 yield
            processed = line.strip()
            if processed:
                yield processed


# ========== 批量处理 ==========

async def batch_process(file_path: str, batch_size: int = 100):
    """分批处理：每批处理 batch_size 条记录后 yield 结果"""
    batch: List[str] = []
    
    async for line in process_large_file_generator(file_path):
        batch.append(line)
        if len(batch) >= batch_size:
            yield batch
            batch = []
    
    # 最后剩余的部分
    if batch:
        yield batch


# ========== FastAPI 流式响应 ==========

app = FastAPI()


@app.get("/download/large-file")
async def download_large_file():
    """流式下载大文件"""
    file_path = "data/large_dataset.json"
    
    async def generate():
        """异步生成器产出 JSON 行"""
        async with aiofiles.open(file_path, "r", encoding="utf-8") as f:
            async for line in f:
                yield line
    
    return StreamingResponse(
        generate(),
        media_type="application/x-ndjson",
        headers={
            "Content-Disposition": "attachment; filename=large_dataset.json",
        }
    )


@app.post("/upload/large-file")
async def upload_large_file(file: UploadFile):
    """分块接收大文件上传"""
    save_path = f"uploads/{file.filename}"
    
    # 流式写入，避免全量加载
    async with aiofiles.open(save_path, "wb") as f:
        while True:
            chunk = await file.read(1024 * 1024)  # 1MB 每块
            if not chunk:
                break
            await f.write(chunk)
    
    file_size = os.path.getsize(save_path)
    return {
        "filename": file.filename,
        "size": file_size,
        "status": "uploaded",
    }


# ========== 支持 Range 的断点续传 ==========

@app.get("/download/range/{file_name}")
async def download_with_range(file_name: str, range: str = ""):
    """支持 Range 头部的断点续传下载"""
    file_path = f"data/{file_name}"
    file_size = os.path.getsize(file_path)
    
    start = 0
    end = file_size - 1
    
    if range:
        # 解析 Range header: bytes=start-end
        range_value = range.replace("bytes=", "")
        parts = range_value.split("-")
        if parts[0]:
            start = int(parts[0])
        if parts[1]:
            end = int(parts[1])
    
    content_length = end - start + 1
    
    async def generate_range():
        async with aiofiles.open(file_path, "rb") as f:
            await f.seek(start)
            remaining = content_length
            while remaining > 0:
                chunk_size = min(8192, remaining)
                chunk = await f.read(chunk_size)
                if not chunk:
                    break
                yield chunk
                remaining -= len(chunk)
    
    return StreamingResponse(
        generate_range(),
        media_type="application/octet-stream",
        status_code=206 if range else 200,
        headers={
            "Content-Range": f"bytes {start}-{end}/{file_size}",
            "Content-Length": str(content_length),
            "Accept-Ranges": "bytes",
        }
    )


# ========== 内存监控 ==========

class MemoryMonitor:
    """内存使用监控"""
    
    def __init__(self, warning_mb: int = 500, limit_mb: int = 1000):
        self.warning_mb = warning_mb
        self.limit_mb = limit_mb
    
    def get_current_memory_mb(self) -> float:
        """获取当前进程内存占用（MB）"""
        import psutil
        process = psutil.Process(os.getpid())
        return process.memory_info().rss / 1024 / 1024
    
    def check_memory(self):
        """检查内存使用，超限时告警"""
        current = self.get_current_memory_mb()
        if current > self.limit_mb:
            raise MemoryError(f"内存超限: {current:.1f}MB > {self.limit_mb}MB")
        if current > self.warning_mb:
            print(f"[Memory] 警告: {current:.1f}MB > {self.warning_mb}MB")


# ========== 安全使用 Generator ==========

def process_in_chunks(data: List[str], chunk_size: int = 1000) -> Iterator[List[str]]:
    """使用 Generator 分批处理列表数据"""
    for i in range(0, len(data), chunk_size):
        yield data[i:i + chunk_size]


async def safe_generator_usage():
    """
    安全使用 Generator 的最佳实践
    """
    # 1. 使用 async for 消费异步生成器
    async for batch in batch_process("large_file.txt"):
        await process_batch(batch)
    
    # 2. 确保生成器正确关闭
    gen = process_large_file_generator("data.txt")
    try:
        async for item in gen:
            if should_stop(item):
                break  # 提前退出时生成器会被 gc 自动关闭
            await process_item(item)
    finally:
        await gen.aclose()  # 显式关闭
```

## 错误示例

```python
# 错误 1：全量加载大文件到内存

def bad_full_file_load():
    """整个文件读到内存"""
    with open("large_file_10gb.txt", "r") as f:
        content = f.read()  # 10GB → OOM!
    # 应使用逐行读取或分块读取


# 错误 2：API 返回大 JSON 不使用流式

@app.get("/bad/download")
async def bad_download():
    """把整个大 JSON 加载到内存再返回"""
    with open("large_data.json", "r") as f:
        data = json.load(f)  # 全部加载到内存
        
    return data  # 序列化为大 JSON 再次占用内存
    # 应使用 StreamingResponse


# 错误 3：上传大文件时全量读取

@app.post("/bad/upload")
async def bad_upload(file: UploadFile):
    """整个上传文件读到内存"""
    content = await file.read()  # 5GB 文件 → OOM!
    with open("save.bin", "wb") as f:
        f.write(content)
    # 应分块读取和写入


# 错误 4：Generator 返回前堆积大量数据

def bad_generator_accumulation():
    """Generator 在 yield 前堆积大量数据"""
    def process(items):
        results = []
        for item in items:
            # 复杂处理，结果不断累积
            processed = expensive_transform(item)
            results.append(processed)
            # 等到所有处理完成才 yield，内存占用持续增长
        yield results  # 太晚 yield
    
    # 正确做法：每处理一条就 yield


# 错误 5：不清理临时文件

def bad_temp_file():
    """临时文件不清理"""
    import tempfile
    
    tmp = tempfile.NamedTemporaryFile(delete=False)
    tmp.write(b"data")
    # 忘记 tmp.close() 或在 finally 中删除
    # 程序结束后留下大量临时文件
```

```python
def read_large_file_random_access(file_path: str):
    """使用 mmap 内存映射大文件"""
    with open(file_path, "r+b") as f:
        with mmap.mmap(f.fileno(), 0) as mm:
```
