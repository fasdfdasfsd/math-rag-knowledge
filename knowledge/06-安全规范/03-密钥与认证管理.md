# 密钥与认证管理

## 概述

密钥管理和认证体系是应用安全的基础。本文件涵盖 API Key 管理原则、环境变量配置、密钥管理服务、JWT 和 OAuth2 最佳实践，以及 RBAC 权限模型在 RAG 项目中的落地方式。

---

## 核心规则
### 🔴 必须遵守 (MUST)

1. **绝不硬编码密钥**：密钥必须从环境变量或密钥管理服务加载
2. **绝不提交密钥到版本控制**：.env 文件必须添加到 .gitignore
3. **使用短有效期凭证**：JWT access token 不超过 15 分钟
4. **最小权限原则**：每个服务/用户仅赋予必要的权限
5. **密钥轮换**：定期轮换密钥，泄露时立即轮换

### 🟡 强烈建议 (SHOULD)

1. 使用非对称加密（RS256/ES256）替代对称加密（HS256）
2. 实施多因素认证（MFA）
3. 使用密钥管理服务（HashiCorp Vault / AWS Secrets Manager）
4. 密钥访问审计日志

### 🟢 可选建议 (MAY)

1. 硬件安全模块（HSM）
2. 零信任架构（Zero Trust Architecture）

## 1. API Key 管理原则

```python
# ===== API Key 管理核心规则 =====
"""
原则 1: 绝不硬编码
  ✅ 环境变量加载
  🚫 代码中写死密钥

原则 2: 绝不提交到 Git
  ✅ .gitignore 包含 .env
  🚫 提交 .env 或含密钥的配置文件

原则 3: 最小权限
  ✅ 每个 API Key 绑定特定权限
  🚫 使用 root 级别的通用密钥

原则 4: 可追溯
  ✅ 每个 Key 关联到具体使用者
  🚫 多人共享同一个 Key
"""

# ===== API Key 生成与验证 =====
import secrets
import hashlib

class APIKeyManager:
    """API Key 安全管理器"""

    @staticmethod
    def generate_api_key() -> tuple[str, str]:
        """
        生成 API Key。
        返回 (plaintext_key, hashed_key)
        仅存储哈希值，明文仅展示一次。
        """
        key = secrets.token_urlsafe(32)  # 32 字节 = 256 位熵
        hashed = hashlib.sha256(key.encode()).hexdigest()
        return key, hashed

    @staticmethod
    def verify_api_key(plaintext_key: str, stored_hash: str) -> bool:
        """验证 API Key"""
        hashed = hashlib.sha256(plaintext_key.encode()).hexdigest()
        return hmac.compare_digest(hashed, stored_hash)

    @staticmethod
    def mask_api_key(key: str) -> str:
        """脱敏 API Key（仅显示前 8 位）"""
        return key[:8] + "****" + key[-4:]
```

### 正确示例：pydantic-settings 加载配置

```python
# config.py
from pydantic_settings import BaseSettings
from pydantic import Field, SecretStr
from typing import Optional


class Settings(BaseSettings):
    """应用配置——所有敏感信息从环境变量加载"""

    # ===== 应用基础配置 =====
    app_name: str = "RAG Application"
    environment: str = "development"  # development / staging / production
    debug: bool = False

    # ===== 密钥配置（使用 SecretStr 防止意外泄露）=====
    jwt_secret_key: SecretStr = Field(..., alias="JWT_SECRET_KEY")
    jwt_algorithm: str = "RS256"
    jwt_access_token_expire_minutes: int = 15

    # ===== 外部服务密钥 =====
    openai_api_key: SecretStr = Field(..., alias="OPENAI_API_KEY")
    chroma_db_host: str = "localhost"
    chroma_db_port: int = 8000

    # ===== 数据库配置 =====
    database_url: SecretStr = Field(..., alias="DATABASE_URL")

    # ===== 密钥管理服务（可选）=====
    vault_addr: Optional[str] = None
    vault_token: Optional[SecretStr] = None

    class Config:
        env_file = ".env"
        env_file_encoding = "utf-8"
        # 禁止未定义字段
        extra = "forbid"

    # ===== 安全验证 =====
    def validate_production(self):
        """生产环境安全检查"""
        if self.environment == "production":
            if self.debug:
                raise ValueError("生产环境不能开启 debug 模式！")
            if not self.jwt_secret_key.get_secret_value():
                raise ValueError("生产环境必须配置 JWT_SECRET_KEY！")
            if not self.database_url.get_secret_value():
                raise ValueError("生产环境必须配置 DATABASE_URL！")


settings = Settings()

# 生产环境验证
if settings.environment == "production":
    settings.validate_production()
```

## 3. Secrets 管理（HashiCorp Vault / AWS Secrets Manager）

```python
# ===== HashiCorp Vault 集成 =====
import hvac
from typing import Optional


class VaultSecretManager:
    """HashiCorp Vault 密钥管理"""

    def __init__(self, vault_addr: str, vault_token: str):
        self.client = hvac.Client(url=vault_addr, token=vault_token)
        if not self.client.is_authenticated():
            raise ConnectionError("Vault 认证失败")

    def get_secret(self, path: str, key: str) -> Optional[str]:
        """从 Vault 读取密钥"""
        try:
            secret = self.client.secrets.kv.v2.read_secret_version(
                path=path, mount_point="secret"
            )
            return secret["data"]["data"].get(key)
        except Exception as e:
            print(f"[ERROR] Vault 读取失败: {e}")
            return None

    def rotate_secret(self, path: str, key: str, new_value: str):
        """轮换密钥"""
        try:
            self.client.secrets.kv.v2.create_or_update_secret(
                path=path,
                secret={key: new_value},
                mount_point="secret",
            )
            print(f"[INFO] 密钥 {path}/{key} 已轮换")
        except Exception as e:
            print(f"[ERROR] 密钥轮换失败: {e}")


# ===== AWS Secrets Manager 集成 =====
import boto3
from botocore.exceptions import ClientError

class AWSSecretManager:
    """AWS Secrets Manager 密钥管理"""

    def __init__(self, region_name: str = "us-east-1"):
        self.client = boto3.client("secretsmanager", region_name=region_name)

    def get_secret(self, secret_name: str) -> dict:
        """获取密钥"""
        try:
            response = self.client.get_secret_value(SecretId=secret_name)
            import json
            return json.loads(response["SecretString"])
        except ClientError as e:
            print(f"[ERROR] AWS Secrets Manager 错误: {e}")
            return {}

    def rotate_secret(self, secret_name: str, secret_string: str):
        """轮换密钥"""
        self.client.put_secret_value(
            SecretId=secret_name,
            SecretString=secret_string,
        )
```

## 4. JWT 最佳实践

```python
from datetime import datetime, timedelta, timezone
from typing import Optional
import jwt
from fastapi import HTTPException, Security
from fastapi.security import HTTPBearer

security = HTTPBearer()


# ===== JWT 安全配置 =====
class JWTSecurityConfig:
    """JWT 安全配置"""

    # 使用非对称算法（推荐）
    ALGORITHM = "RS256"

    # 短有效期
    ACCESS_TOKEN_EXPIRE_MINUTES = 15
    REFRESH_TOKEN_EXPIRE_DAYS = 7

    # Token 类型
    TOKEN_TYPE_ACCESS = "access"
    TOKEN_TYPE_REFRESH = "refresh"

    # 必须包含的声明
    REQUIRED_CLAIMS = {"sub", "exp", "iat", "type"}


# ===== JWT 令牌管理 =====
class JWTManager:
    """JWT 令牌安全管理器"""

    def __init__(self, private_key: str, public_key: str):
        # 生产环境应使用 RSA 密钥对
        self.private_key = private_key
        self.public_key = public_key

    def create_access_token(
        self,
        subject: str,
        extra_claims: Optional[dict] = None,
    ) -> str:
        """创建访问令牌（短有效期）"""
        now = datetime.now(timezone.utc)
        payload = {
            "sub": subject,
            "iat": now,
            "exp": now + timedelta(
                minutes=JWTSecurityConfig.ACCESS_TOKEN_EXPIRE_MINUTES
            ),
            "type": JWSTokenConfig.TOKEN_TYPE_ACCESS,
            "iss": "rag-app",
        }
        if extra_claims:
            payload.update(extra_claims)
        return jwt.encode(payload, self.private_key, algorithm=JWTSecurityConfig.ALGORITHM)

    def create_refresh_token(self, subject: str) -> str:
        """创建刷新令牌"""
        now = datetime.now(timezone.utc)
        payload = {
            "sub": subject,
            "iat": now,
            "exp": now + timedelta(days=JWTSecurityConfig.REFRESH_TOKEN_EXPIRE_DAYS),
            "type": JWTSecurityConfig.TOKEN_TYPE_REFRESH,
            "iss": "rag-app",
        }
        return jwt.encode(payload, self.private_key, algorithm=JWTSecurityConfig.ALGORITHM)

    def verify_token(self, token: str) -> dict:
        """验证 Token"""
        try:
            payload = jwt.decode(
                token,
                self.public_key,
                algorithms=[JWTSecurityConfig.ALGORITHM],
                # 以下选项增强安全性
                options={
                    "require": ["sub", "exp", "iat", "type"],
                    "verify_exp": True,
                    "verify_iat": True,
                },
            )

            # 检查 Token 类型
            token_type = payload.get("type")
            if token_type not in [
                JWTSecurityConfig.TOKEN_TYPE_ACCESS,
                JWTSecurityConfig.TOKEN_TYPE_REFRESH,
            ]:
                raise HTTPException(status_code=401, detail="无效 Token 类型")

            return payload

        except jwt.ExpiredSignatureError:
            raise HTTPException(status_code=401, detail="Token 已过期")
        except jwt.InvalidTokenError as e:
            raise HTTPException(status_code=401, detail=f"无效 Token: {e}")


# ===== FastAPI 依赖注入 =====
from fastapi import Depends
from fastapi.security import HTTPAuthorizationCredentials

async def get_current_user(
    credentials: HTTPAuthorizationCredentials = Security(security),
) -> dict:
    """获取当前认证用户（FastAPI 依赖注入）"""
    token = credentials.credentials
    jwt_manager = JWTManager(private_key, public_key)  # 实际应从单例获取
    payload = jwt_manager.verify_token(token)
    return payload


@app.get("/api/v1/me")
async def get_me(current_user: dict = Depends(get_current_user)):
    """获取当前用户信息（需要 JWT 认证）"""
    return {"user_id": current_user["sub"]}
```

## 5. OAuth2 流程安全

```python
from fastapi import FastAPI, HTTPException, Request
from authlib.integrations.starlette_client import OAuth

# ===== OAuth2 安全配置 =====

class OAuth2Security:
    """OAuth2 安全实践"""

    @staticmethod
    def validate_state_param(request: Request, state: str) -> bool:
        """验证 state 参数防止 CSRF"""
        saved_state = request.session.get("oauth_state")
        if not saved_state or saved_state != state:
            raise HTTPException(status_code=400, detail="State 参数无效（CSRF 攻击防范）")
        return True

    @staticmethod
    def validate_redirect_uri(redirect_uri: str) -> bool:
        """白名单校验重定向 URI"""
        allowed_uris = {
            "https://app.example.com/callback",
            "http://localhost:8000/callback",  # 开发环境
        }
        if redirect_uri not in allowed_uris:
            raise HTTPException(status_code=400, detail="未授权的重定向 URI")
        return True

    @staticmethod
    def use_pkce(client: OAuth):
        """强制使用 PKCE（Proof Key for Code Exchange）"""
        # PKCE 防止授权码拦截攻击
        client.register_client_auth_method("PKCE")


# ===== OAuth2 授权码模式 =====
oauth = OAuth()
oauth.register(
    name="auth0",
    client_id="your-client-id",
    client_secret="your-client-secret",
    server_metadata_url="https://your-tenant.auth0.com/.well-known/openid-configuration",
    client_kwargs={
        "scope": "openid profile email",
        "code_challenge_method": "S256",  # PKCE 强制
    },
)
```

## 6. RBAC 权限模型

```python
from enum import Enum
from typing import Set, Optional


# ===== 权限定义 =====
class Permission(str, Enum):
    """系统权限枚举"""

    # 文档权限
    DOCUMENT_CREATE = "document:create"
    DOCUMENT_READ = "document:read"
    DOCUMENT_UPDATE = "document:update"
    DOCUMENT_DELETE = "document:delete"

    # 用户管理
    USER_CREATE = "user:create"
    USER_READ = "user:read"
    USER_UPDATE = "user:update"
    USER_DELETE = "user:delete"

    # 系统管理
    SYSTEM_CONFIG = "system:config"
    SYSTEM_LOGS = "system:logs"
    SYSTEM_METRICS = "system:metrics"


# ===== 角色定义 =====
class Role(str, Enum):
    """角色枚举——每个角色绑定一组权限"""
    ADMIN = "admin"         # 系统管理员
    EDITOR = "editor"       # 内容编辑
    REVIEWER = "reviewer"   # 内容审核
    VIEWER = "viewer"       # 只读用户
    API = "api"             # API 调用（机器账号）


# ===== 角色-权限映射 =====
ROLE_PERMISSIONS: dict[Role, Set[Permission]] = {
    Role.ADMIN: {
        Permission.DOCUMENT_CREATE,
        Permission.DOCUMENT_READ,
        Permission.DOCUMENT_UPDATE,
        Permission.DOCUMENT_DELETE,
        Permission.USER_CREATE,
        Permission.USER_READ,
        Permission.USER_DELETE,
        Permission.SYSTEM_CONFIG,
        Permission.SYSTEM_LOGS,
        Permission.SYSTEM_METRICS,
    },
    Role.EDITOR: {
        Permission.DOCUMENT_CREATE,
        Permission.DOCUMENT_READ,
        Permission.DOCUMENT_UPDATE,
    },
    Role.REVIEWER: {
        Permission.DOCUMENT_READ,
        Permission.DOCUMENT_UPDATE,  # 仅审核状态
    },
    Role.VIEWER: {
        Permission.DOCUMENT_READ,
    },
    Role.API: {
        Permission.DOCUMENT_READ,
        Permission.SYSTEM_METRICS,
    },
}


# ===== 权限校验器 =====
class RBACAuthorizer:
    """基于角色的访问控制"""

    @staticmethod
    def has_permission(
        user_roles: list[Role],
        required_permission: Permission,
    ) -> bool:
        """检查用户是否有指定权限"""
        for role in user_roles:
            permissions = ROLE_PERMISSIONS.get(role, set())
            if required_permission in permissions:
                return True
        return False

    @staticmethod
    def require_permission(permission: Permission):
        """权限校验装饰器"""
        from functools import wraps

        def decorator(func):
            @wraps(func)
            async def wrapper(*args, **kwargs):
                user = get_current_user()  # 从请求上下文中获取
                if not RBACAuthorizer.has_permission(user.roles, permission):
                    raise HTTPException(
                        status_code=403,
                        detail=f"权限不足: 需要 {permission.value}",
                    )
                return await func(*args, **kwargs)
            return wrapper
        return decorator


# ===== 使用示例 =====
@app.get("/api/v1/documents")
@RBACAuthorizer.require_permission(Permission.DOCUMENT_READ)
async def list_documents():
    """列出文档——需要读取权限"""
    return get_all_documents()


@app.post("/api/v1/documents")
@RBACAuthorizer.require_permission(Permission.DOCUMENT_CREATE)
async def create_document():
    """创建文档——需要写入权限"""
    return create_new_document()
```

## 7. pre-commit Hook 检测密钥泄露

```yaml
# .pre-commit-config.yaml
repos:
  # 密钥泄露检测
  - repo: https://github.com/gitleaks/gitleaks
    rev: v8.18.0

      - id: gitleaks
        args:
          - "--verbose"
          - "--redact"

  # 检测高熵字符串（可能是密钥）
  - repo: https://github.com/pre-commit/pre-commit-hooks
    rev: v4.6.0
    hooks:
      - id: detect-private-key
      - id: check-added-large-files  # 防止意外提交大文件

  # 自定义 hook：检查 .env 文件是否被暂存
  - repo: local
    hooks:
      - id: check-env-files
        name: Check .env files not staged
        entry: bash -c 'git diff --cached --name-only | grep -E "^\.env" && exit 1 || exit 0'
        language: system
```

## 错误示例集锦

### 🚫 错误：代码中硬编码 Secret Key（极其危险）

```python

# 所有能看到代码的人都可以获取该密钥！

JWT_SECRET = "jwt-secret-key-67890"

# 提交到 Git 后，密钥永久暴露在版本历史中
# 即使删除文件，历史记录仍可查到
```

### 🚫 错误：.env 文件提交到版本控制

```bash
# 🚫 错误：没有添加 .env 到 .gitignore
# 结果：git push 后 .env 文件中的密钥全部泄露

# ✅ 正确做法：
# 1. 在 .gitignore 中添加 .env
# 2. 使用 .env.example 作为模板
# 3. 使用 git-secrets / gitleaks 扫描历史
```

### 🚫 错误：JWT 使用弱算法和弱密钥

```python
# 🚫 错误：使用 HS256 + 弱密钥
token = jwt.encode(
    {"sub": user.id},
    "weak_secret",  # 可被暴力破解
    algorithm="HS256"  # 对称密钥，泄露后攻击者可伪造任意 Token
)

# ✅ 正确：使用 RS256（非对称）
import jwt
from cryptography.hazmat.primitives import serialization

# 使用 RSA 密钥对
with open("private.pem", "rb") as f:
    private_key = serialization.load_pem_private_key(
        f.read(), password=None
    )
token = jwt.encode(
    {"sub": user.id},
    private_key,
    algorithm="RS256",
)
```

```python
# 🚫 错误：日志记录敏感信息

# ✅ 正确：日志脱敏
```
