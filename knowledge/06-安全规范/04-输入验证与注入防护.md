# 输入验证与注入防护

## 概述

输入验证是应用安全的第一道防线。未经检验的输入是注入攻击（SQL 注入、XSS、命令注入、路径遍历等）的共同根源。本文件覆盖 RAG 项目（API 层、检索层、文档处理层）常见的注入攻击类型及防护措施，重点使用 Python/Pydantic/FastAPI 的实践方案。

---

## 核心规则
### 🔴 必须遵守 (MUST)

1. **所有外部输入不可信**：用户输入、API 请求、文件上传、外部 API 回调数据全部视为不可信
2. **allowlist 优先**：定义允许的内容模式，拒绝一切不在允许列表中的输入
3. **参数化查询**：所有数据库查询必须使用参数化或 ORM，禁用字符串拼接
4. **输出编码**：LLM 输出渲染到 Web 页面必须进行 HTML 编码

### 🟡 强烈建议 (SHOULD)

1. 使用 Pydantic 进行请求体校验
2. 实施输入长度限制
3. 使用 Content Security Policy（CSP）头
4. 文件上传限制类型和大小

### 🟢 可选建议 (MAY)

1. Web 应用防火墙（WAF）
2. 请求签名验证

## 1. SQL 注入防护

### 正确示例：参数化查询和 ORM 使用

```python
# ===== 方法一：参数化查询（推荐）=====
import sqlite3
from typing import Optional

class UserRepository:
    """用户仓储——安全的数据访问"""

    def __init__(self, db_path: str):
        self.conn = sqlite3.connect(db_path)

    def get_user_by_id(self, user_id: int) -> Optional[dict]:
        """
        安全：使用 ? 占位符的参数化查询。
        参数由数据库驱动转义，杜绝注入。
        """
        cursor = self.conn.cursor()
        cursor.execute(
            "SELECT id, username, email, created_at FROM users WHERE id = ?",
            (user_id,),  # 参数化——安全！
        )
        row = cursor.fetchone()
        return dict(row) if row else None

    def get_users_by_role(self, role: str, limit: int = 10) -> list[dict]:
        """安全：参数化查询（多个参数）"""
        cursor = self.conn.cursor()
        cursor.execute(
            "SELECT id, username, role FROM users WHERE role = ? ORDER BY id LIMIT ?",
            (role, limit),
        )
        return [dict(row) for row in cursor.fetchall()]

    def search_users_safe(self, keyword: str) -> list[dict]:
        """安全：LIKE 查询也使用参数化"""
        cursor = self.conn.cursor()
        # 注意：LIKE 通配符需要在 Python 层面处理，而不是 SQL 拼接
        search_term = f"%{keyword}%"
        cursor.execute(
            "SELECT id, username FROM users WHERE username LIKE ?",
            (search_term,),
        )
        return [dict(row) for row in cursor.fetchall()]


# ===== 方法二：ORM（SQLAlchemy）=====
from sqlalchemy import create_engine, Column, Integer, String, text
from sqlalchemy.orm import declarative_base, Session

Base = declarative_base()

class User(Base):
    __tablename__ = "users"
    id = Column(Integer, primary_key=True)
    username = Column(String, nullable=False, unique=True)
    email = Column(String)

class UserService:
    def __init__(self, engine):
        self.engine = engine

    def get_user_safe(self, session: Session, user_id: int) -> Optional[User]:
        """安全：ORM 自动参数化"""
        return session.get(User, user_id)

    def get_user_by_name(self, session: Session, name: str) -> Optional[User]:
        """安全：ORM 查询参数自动转义"""
        return session.query(User).filter(User.username == name).first()

    def raw_query_safe(self, session: Session, user_id: int) -> list[dict]:
        """安全：text() + 命名参数（即使写原生 SQL 也安全）"""
        result = session.execute(
            text("SELECT * FROM users WHERE id = :user_id"),
            {"user_id": user_id},
        )
        return [row._mapping for row in result]
```

```python
    """Content Security Policy 中间件"""
```

## 3. CSRF 防护

### 正确示例：SameSite Cookie + CSRF Token

```python
from fastapi import FastAPI, Request, HTTPException
from fastapi.middleware.cors import CORSMiddleware
from starlette.middleware.sessions import SessionMiddleware
import secrets


app = FastAPI()

# ===== SameSite Cookie 配置 =====
from starlette.responses import Response

class SecureCookieMiddleware(BaseHTTPMiddleware):
    async def dispatch(self, request: Request, call_next):
        response: Response = await call_next(request)

        # 为所有 Cookie 设置安全属性
        for key, value in request.cookies.items():
            response.set_cookie(
                key=key,
                value=value,
                httponly=True,      # 禁止 JS 访问
                secure=True,         # 仅 HTTPS
                samesite="lax",     # 防范 CSRF
                max_age=3600,
            )
        return response


# ===== CSRF Token 验证 =====
class CSRFProtection:
    """CSRF 防护工具"""

    @staticmethod
    def generate_token() -> str:
        """生成 CSRF Token"""
        return secrets.token_urlsafe(32)

    @staticmethod
    def validate_token(request: Request, token: str) -> bool:
        """验证 CSRF Token"""
        expected_token = request.session.get("csrf_token")
        if not expected_token or expected_token != token:
            raise HTTPException(status_code=403, detail="CSRF Token 无效")
        return True


# ===== CORS 配置（辅助 CSRF 防护）=====
app.add_middleware(
    CORSMiddleware,
    allow_origins=["https://app.example.com"],  # 不写 ["*"]
    allow_credentials=True,
    allow_methods=["GET", "POST"],
    allow_headers=["Authorization", "Content-Type", "X-CSRF-Token"],
)

# 添加 Session 中间件（用于存储 CSRF Token）
app.add_middleware(SessionMiddleware, secret_key="your-secret-key")
app.add_middleware(SecureCookieMiddleware)


@app.post("/api/v1/documents")
async def create_document(
    request: Request,
    csrf_token: str = Header(...),
):
    """创建文档——需要 CSRF Token 验证"""
    CSRFProtection.validate_token(request, csrf_token)
    return create_document_in_db()
```

## 4. 命令注入防护

### 正确示例：禁止 os.system()，使用 subprocess 参数列表

```python
import subprocess
import shlex
from typing import List


# ===== 安全的命令执行 =====

class SafeCommandExecutor:
    """安全的命令执行器"""

    # 允许的命令白名单
    ALLOWED_COMMANDS = {
        "git",
        "python",
        "pip",
        "node",
    }

    @classmethod
    def execute(cls, command: str, args: List[str], timeout: int = 30) -> str:
        """
        安全执行命令。
        使用参数列表形式，不通过 shell 解析。
        """
        if command not in cls.ALLOWED_COMMANDS:
            raise ValueError(f"命令 '{command}' 不在白名单中")

        # 🚫 不要这样做：subprocess.run(f"{command} {args}", shell=True)
        # ✅ 正确：传递参数列表
        result = subprocess.run(
            [command, *args],
            capture_output=True,
            text=True,
            timeout=timeout,
            # 最关键：不开启 shell！
            shell=False,
        )

        if result.returncode != 0:
            raise RuntimeError(f"命令执行失败: {result.stderr}")

        return result.stdout


# ===== 路径白名单校验 =====
class PathValidator:
    """路径安全校验"""

    ALLOWED_BASE_DIRS = {
        "/app/data",
        "/app/output",
    }

    @classmethod
    def validate_path(cls, user_path: str) -> str:
        """验证路径安全，防止路径遍历"""
        import os

        # 1. 规范化路径
        normalized = os.path.normpath(os.path.join("/app/data", user_path))

        # 2. 检查是否在允许的基目录下
        safe = False
        for base_dir in cls.ALLOWED_BASE_DIRS:
            if normalized.startswith(base_dir):
                safe = True
                break

        if not safe:
            raise ValueError(f"路径 '{user_path}' 不在允许的目录范围内")

        return normalized
```

## 5. 输入验证：Pydantic + Allowlist 策略

### 正确示例：Pydantic 验证器

```python
from pydantic import BaseModel, Field, field_validator, EmailStr
from typing import Optional, List
import re


# ===== 严格的输入模型 =====
class DocumentCreateRequest(BaseModel):
    """文档创建请求——严格的输入验证"""

    title: str = Field(
        ...,
        min_length=1,
        max_length=200,
        description="文档标题，1-200 字符",
    )
    content: str = Field(
        ...,
        min_length=1,
        max_length=100000,
        description="文档内容，最大 100KB",
    )
    tags: Optional[List[str]] = Field(
        default=None,
        max_length=10,
        description="标签，最多 10 个",
    )
    visibility: str = Field(
        default="private",
        pattern=r"^(public|private|team)$",
        description="可见性：public/private/team",
    )

    # 自定义验证器：标题不允许 HTML 标签
    @field_validator("title")
    @classmethod
    def sanitize_title(cls, v: str) -> str:
        # 去除多余的空白
        v = " ".join(v.split())
        # 检测和禁止 HTML 标签
        if re.search(r"<[^>]*>", v):
            raise ValueError("标题不能包含 HTML 标签")
        return v

    # 自定义验证器：content 清洗
    @field_validator("content")
    @classmethod
    def sanitize_content(cls, v: str) -> str:
        # 去除控制字符（保留换行）
        v = re.sub(r"[\x00-\x08\x0B\x0C\x0E-\x1F]", "", v)
        return v

    # 自定义验证器：标签清洗
    @field_validator("tags")
    @classmethod
    def sanitize_tags(cls, v: Optional[List[str]]) -> Optional[List[str]]:
        if v is None:
            return v
        cleaned = []
        for tag in v:
            # 只允许字母、数字、下划线、连字符
            if not re.match(r"^[a-zA-Z0-9_-]{1,30}$", tag):
                raise ValueError(f"标签 '{tag}' 包含非法字符")
            cleaned.append(tag)
        return list(set(cleaned))  # 去重
```

## 6. 路径遍历防护

### 正确示例：

```python
import os
from pathlib import Path

class FileService:
    """安全的文件服务——防止路径遍历"""

    # 允许用户访问的基目录
    BASE_DIR = Path("/app/data/documents")

    @classmethod
    def get_safe_path(cls, user_input_path: str) -> Path:
        """
        将用户输入转换为安全的文件路径。

        1. 拒绝包含 '..' 的路径
        2. 规范化路径
        3. 验证在基目录内
        """
        # 检查路径遍历尝试
        if ".." in user_input_path:
            raise ValueError("路径中不能包含 '..'")

        # 规范化路径
        safe_path = (cls.BASE_DIR / user_input_path).resolve()

        # 验证路径在 BASE_DIR 内
        if not str(safe_path).startswith(str(cls.BASE_DIR.resolve())):
            raise ValueError("路径超出允许的访问范围")

        return safe_path

    @classmethod
    def read_file(cls, relative_path: str) -> str:
        """安全地读取文件"""
        safe_path = cls.get_safe_path(relative_path)

        if not safe_path.exists():
            raise FileNotFoundError(f"文件不存在: {relative_path}")

        if not safe_path.is_file():
            raise ValueError(f"不是文件: {relative_path}")

        return safe_path.read_text(encoding="utf-8")


# ===== FastAPI 文件上传安全 =====
from fastapi import UploadFile, File

ALLOWED_EXTENSIONS = {".txt", ".md", ".pdf", ".json", ".csv"}
MAX_FILE_SIZE = 10 * 1024 * 1024  # 10 MB

async def safe_file_upload(file: UploadFile = File(...)):
    """安全的文件上传"""
    # 1. 检查文件扩展名
    extension = os.path.splitext(file.filename)[1].lower()
    if extension not in ALLOWED_EXTENSIONS:
        raise HTTPException(status_code=400, detail=f"不支持的文件类型: {extension}")

    # 2. 检查文件名安全性
    filename = os.path.basename(file.filename)  # 去除路径
    if ".." in filename or "/" in filename or "\\" in filename:
        raise HTTPException(status_code=400, detail="非法的文件名")

    # 3. 检查文件大小
    content = await file.read()
    if len(content) > MAX_FILE_SIZE:
        raise HTTPException(status_code=400, detail=f"文件过大，最大 {MAX_FILE_SIZE // 1024 // 1024}MB")

    return filename, content
```

```python
        ...,

                raise ValueError(f"包含过长的词（{len(word)}字符）")

    @field_validator("document_ids")
    @classmethod
    def validate_document_ids(cls, v: list[str]) -> list[str]:
        # 只允许 UUID 格式
        uuid_pattern = re.compile(
            r"^[0-9a-f]{8}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{4}-[0-9a-f]{12}$"
        )
        for doc_id in v:
            if not uuid_pattern.match(doc_id):
                raise ValueError(f"文档 ID 格式无效: {doc_id}")
        return v


@app.post("/api/v1/rag/query")
async def rag_query(
    request: RAGQueryRequest,
    user: User = Depends(get_current_user),
):
    """RAG 查询——安全校验完整链"""

    # 1. 输入验证（Pydantic 已完成）
    query = request.query

    # 2. 权限检查
    authorized_ids = user.filter_authorized_docs(request.document_ids)

    # 3. 检索（参数化——ORM 层）
    documents = retrieve_documents(authorized_ids, query)

    # 4. 输出编码
    safe_documents = []
    for doc in documents:
        safe_documents.append({
            "id": doc.id,
            "content": html.escape(doc.content),
            "score": doc.score,
        })

    # 5. LLM 调用安全包装
    response = safe_llm_call(query, safe_documents)

    return {
        "response": response,
        "sources": safe_documents,
    }
```
