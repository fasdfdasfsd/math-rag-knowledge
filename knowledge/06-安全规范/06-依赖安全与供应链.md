# 依赖安全与供应链

## 概述

现代 Python 项目依赖大量第三方包，供应链攻击已成为最严重的安全威胁之一。本文件覆盖依赖漏洞扫描、SBOM 管理、许可证合规、依赖锁定、供应链攻击防护等关键实践。

> RAG 项目的依赖链尤其复杂：LangChain、ChromaDB、Embedding 模型、LLM SDK 等数十个库构成依赖树，任一环节存在漏洞都可能被利用。

---

## 核心规则
### 🔴 必须遵守 (MUST)

1. **依赖锁定**：使用 uv.lock 或 requirements.txt（精确版本号），禁止使用版本范围（如 `>=`）
2. **漏洞扫描**：每次 CI 构建必须运行 `pip-audit` 扫描
3. **最小依赖**：仅引入必要的依赖，移除未使用的包
4. **依赖来源验证**：只从官方 PyPI 或可信镜像安装
5. **安全更新**：已知漏洞依赖必须在 48 小时内修复

### 🟡 强烈建议 (SHOULD)

1. 生成和维护 SBOM（Software Bill of Materials）
2. 定期（每月）进行全面依赖审计
3. 许可证合规检查
4. 使用 Dependabot / Renovate 自动更新依赖

### 🟢 可选建议 (MAY)

1. 私有 PyPI 镜像缓存
2. 依赖签名验证（PEP 458/480）
3. 运行时依赖完整性校验

## 1. pip audit / safety 依赖漏洞扫描

```python
# ===== pip-audit 使用 =====
"""
# 安装
pip install pip-audit

# 扫描 requirements.txt
pip-audit --requirement requirements.txt --desc on

# 扫描当前环境的已安装包
pip-audit

# JSON 格式输出
pip-audit --requirement requirements.txt --format json

# 忽略特定漏洞（仅在确认不影响时）
pip-audit --requirement requirements.txt \
    --ignore-vuln PYSEC-2023-123 \
    --ignore-vuln PYSEC-2023-456

# CI 模式（漏洞存在则退出码非 0）
pip-audit --requirement requirements.txt --strict
"""


# ===== safety 使用 =====
"""
# 安装
pip install safety

# 扫描
safety check --file requirements.txt

# 详细输出
safety check --file requirements.txt --full-report

# CI 模式
safety check --file requirements.txt --policy-file .safety-policy.yml
"""


# ===== 自动化扫描脚本 =====
import subprocess
import sys

def run_security_scan():
    """运行安全扫描，漏洞存在则终止"""
    results = subprocess.run(
        ["pip-audit", "--requirement", "requirements.txt", "--desc", "on", "--strict"],
        capture_output=True,
        text=True,
    )

    if results.returncode != 0:
        print("=" * 60)
        print("  [!] 依赖安全扫描发现漏洞！")
        print("=" * 60)
        print(results.stdout)
        print(results.stderr)
        sys.exit(1)
    else:
        print("[OK] 依赖安全扫描通过，未发现已知漏洞")


# ===== pyproject.toml 配置 =====
"""
[tool.pip-audit]
strict = true
desc = true
# 在 CI 中跳过 dev 依赖扫描
# skip-dev = true
"""
```

## 2. SBOM（Software Bill of Materials）管理

```python
# ===== 生成 SPDX/CycloneDX 格式 SBOM =====

"""
# 使用 cyclonedx-bom（推荐）
pip install cyclonedx-bom

# 生成 CycloneDX 格式 SBOM
cyclonedx-py -r -o sbom.xml

# 生成 JSON 格式
cyclonedx-py -r --format json -o sbom.json

# Pipenv 项目
cyclonedx-py -p -o sbom.xml
"""


# ===== SBOM 验证脚本 =====
import json
from typing import List, Dict
from datetime import datetime


class SBOMAnalyzer:
    """SBOM 分析与审计"""

    def __init__(self, sbom_path: str):
        with open(sbom_path, "r") as f:
            self.sbom = json.load(f)

    def get_package_list(self) -> List[Dict]:
        """获取所有依赖包列表"""
        if self.sbom.get("bomFormat") == "CycloneDX":
            return self.sbom.get("components", [])
        elif self.sbom.get("spdxVersion"):
            return self.sbom.get("packages", [])
        return []

    def check_outdated_packages(
        self,
        max_age_days: int = 365,
    ) -> List[Dict]:
        """检查超过指定天数未更新的包"""
        outdated = []
        for pkg in self.get_package_list():
            release_date = pkg.get("releaseDate", "")
            if release_date:
                release_dt = datetime.fromisoformat(release_date)
                age = (datetime.now() - release_dt).days
                if age > max_age_days:
                    outdated.append({
                        "name": pkg.get("name"),
                        "version": pkg.get("version"),
                        "age_days": age,
                    })
        return outdated

    def get_dependency_tree(self) -> Dict[str, List[str]]:
        """获取依赖树"""
        tree = {}
        for pkg in self.get_package_list():
            name = f"{pkg.get('name')}=={pkg.get('version')}"
            # 注意：SBOM 格式不同，依赖信息字段不同
            deps = pkg.get("dependencies", []) or pkg.get("relationships", [])
            tree[name] = deps
        return tree

    def generate_report(self) -> str:
        """生成 SBOM 分析报告"""
        packages = self.get_package_list()
        outdated = self.check_outdated_packages()

        report = [
            "=== SBOM 分析报告 ===",
            f"生成时间: {datetime.now().isoformat()}",
            f"总依赖数: {len(packages)}",
            f"过时依赖: {len(outdated)}",
            "",
            "过时依赖列表:",
        ]

        for pkg in outdated:
            report.append(f"  - {pkg['name']}=={pkg['version']} ({pkg['age_days']}天未更新)")

        return "\n".join(report)


# ===== CI/CD 集成 SBOM 生成 =====
"""
# .github/workflows/sbom.yml
name: Generate SBOM
on:
  push:
    branches: [main]
  schedule:
    - cron: "0 0 1 * *"  # 每月1日生成

jobs:
  sbom:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-python@v5
      - name: Generate SBOM
        run: |
          pip install cyclonedx-bom
          cyclonedx-py -r --format json -o sbom.json
      - name: Upload SBOM
        uses: actions/upload-artifact@v4
        with:
          name: sbom
          path: sbom.json
"""
```

```python
output-file = "licenses-report.csv"
format = "csv"
"""


# ===== 许可证审计脚本 =====
import subprocess
import json
from typing import List, Dict

class LicenseAuditor:
    """许可证合规审计"""

    ALLOWED_LICENSES = {
        "MIT", "Apache-2.0", "BSD-3-Clause", "BSD-2-Clause",
        "MPL-2.0", "LGPL-3.0-only", "LGPL-2.1-only",
        "ISC", "Python-2.0", "Unlicense",
    }

    RESTRICTED_LICENSES = {
        "GPL-3.0-only", "GPL-2.0-only",
        "AGPL-3.0-only", "AGPL-2.0-only",
    }

    @classmethod
    def audit(cls) -> Dict[str, List[str]]:
        """审计依赖许可证"""
        result = subprocess.run(
            ["pip-licenses", "--format=json"],
            capture_output=True,
            text=True,
        )
        packages = json.loads(result.stdout)

        issues = {
            "restricted": [],
            "unknown": [],
            "allowed": [],
        }

        for pkg in packages:
            name = pkg.get("Name", "unknown")
            version = pkg.get("Version", "?")
            license = pkg.get("License", "UNKNOWN")
            identifier = f"{name}=={version} ({license})"

            if license in cls.RESTRICTED_LICENSES:
                issues["restricted"].append(identifier)
            elif license not in cls.ALLOWED_LICENSES:
                issues["unknown"].append(identifier)
            else:
                issues["allowed"].append(identifier)

        return issues
```

## 5. 定期更新策略

```python
# ===== 依赖更新策略 =====

class DependencyUpdatePolicy:
    """依赖更新策略"""

    # 修复策略
    URGENT_FIX = "urgent"       # 已知漏洞：24-48小时内修复
    SCHEDULED = "scheduled"     # 常规更新：月更新
    DEFERRED = "deferred"       # 策略性延迟：季度更新

    @staticmethod
    def classify_update(package: str, current: str, latest: str, advisory: bool) -> str:
        """分类更新优先级"""
        if advisory:
            return DependencyUpdatePolicy.URGENT_FIX
        # Major 版本更新
        if current.split(".")[0] != latest.split(".")[0]:
            return DependencyUpdatePolicy.DEFERRED
        return DependencyUpdatePolicy.SCHEDULED


# ===== Dependabot / Renovate 配置 =====

"""
# .github/dependabot.yml
version: 2
updates:
  - package-ecosystem: "pip"
    directory: "/"
    schedule:
      interval: "weekly"
    # 限制每日 PR 数量
    open-pull-requests-limit: 10
    # 分配 Reviewer
    reviewers:
      - "team-security"
    # 添加标签
    labels:
      - "dependencies"
      - "security"
    # 安全更新自动合并
    allow:
      - dependency-type: "production"
    # 忽略主版本更新（需要手动评估）
    ignore:
      - dependency-name: "*"
        update-types: ["version-update:semver-major"]
"""


# ===== 更新检查脚本 =====
import subprocess
import json
from datetime import datetime

def check_outdated_packages():
    """检查过时的包"""
    result = subprocess.run(
        ["pip", "list", "--outdated", "--format=json"],
        capture_output=True,
        text=True,
    )
    outdated = json.loads(result.stdout)

    report = [f"=== 过时依赖检查 ({datetime.now().date()}) ==="]
    report.append(f"总计: {len(outdated)} 个包可更新")

    for pkg in sorted(outdated, key=lambda x: x["name"]):
        report.append(
            f"  {pkg['name']}: {pkg['version']} → {pkg['latest_version']}"
        )

    return "\n".join(report)
```

## 6. 供应链攻击防范

```python
import re
import requests
from typing import List, Tuple, Optional


# ===== Typosquatting 检测 =====
class TyposquattingDetector:
    """检测恶意的包名模仿攻击（Typosquatting）"""

    # 常见的被模仿的高知名度包
    POPULAR_PACKAGES = {
        "requests", "numpy", "pandas", "scipy", "django",
        "flask", "fastapi", "pydantic", "sqlalchemy", "pytest",
        "celery", "redis", "pillow", "matplotlib", "scikit-learn",
        "tensorflow", "torch", "transformers", "langchain", "chromadb",
        "httpx", "beautifulsoup4", "uvicorn", "gunicorn",
    }

    @classmethod
    def check_typosquatting(cls, package_name: str) -> Optional[str]:
        """检查包名是否为 typo-squatting"""
        name_lower = package_name.lower()

        for popular in cls.POPULAR_PACKAGES:
            # 编辑距离 = 1 的排列（常见 typo 模式）
            if cls._is_close_match(name_lower, popular):
                if name_lower != popular:
                    return f"'{package_name}' 疑似模仿 '{popular}'"

            # 常见 typo 模式
            if cls._check_common_typos(name_lower, popular):
                return f"'{package_name}' 疑似模仿 '{popular}'（常见 typo 模式）"

        return None

    @classmethod
    def _is_close_match(cls, a: str, b: str, max_edits: int = 1) -> bool:
        """编辑距离计算"""
        if abs(len(a) - len(b)) > max_edits:
            return False

        # 简化的莱文斯坦距离
        dp = list(range(len(b) + 1))
        for i in range(1, len(a) + 1):
            new_dp = [i] + [0] * len(b)
            for j in range(1, len(b) + 1):
                cost = 0 if a[i-1] == b[j-1] else 1
                new_dp[j] = min(
                    dp[j] + 1,           # 删除
                    new_dp[j-1] + 1,     # 插入
                    dp[j-1] + cost,      # 替换
                )
            dp = new_dp

        return dp[-1] <= max_edits

    @classmethod
    def _check_common_typos(cls, a: str, b: str) -> bool:
        """检查常见的 typo 模式"""
        # 字母交换
        for i in range(len(a) - 1):
            swapped = a[:i] + a[i+1] + a[i] + a[i+2:]
            if swapped == b:
                return True

        # 重复字母
        if re.search(r"(.)\1", a):
            dedup = re.sub(r"(.)\1+", r"\1", a)
            if dedup == b:
                return True

        # 缺失字母
        for i in range(len(a)):
            removed = a[:i] + a[i+1:]
            if removed == b:
                return True

        return False


# ===== 包来源验证 =====
class PackageSourceVerifier:
    """验证包来源的可靠性"""

    TRUSTED_DOMAINS = {
        "pypi.org",
        "files.pythonhosted.org",
        "pypi.python.org",
    }

    @classmethod
    def verify_package_url(cls, url: str) -> bool:
        """验证下载URL是否为官方来源"""
        from urllib.parse import urlparse
        parsed = urlparse(url)
        return parsed.hostname in cls.TRUSTED_DOMAINS

    @classmethod
    def check_package_info(cls, package_name: str) -> dict:
        """查询 PyPI 上的包信息"""
        response = requests.get(
            f"https://pypi.org/pypi/{package_name}/json",
            timeout=10,
        )
        if response.status_code != 200:
            return {"exists": False, "message": f"包 '{package_name}' 在 PyPI 上不存在"}

        data = response.json()
        info = data.get("info", {})

        return {
            "exists": True,
            "name": info.get("name"),
            "author": info.get("author"),
            "author_email": info.get("author_email"),
            "latest_version": info.get("version"),
            "home_page": info.get("home_page"),
            "project_urls": info.get("project_urls"),
            "release_count": len(data.get("releases", {})),
            "is_new_package": info.get("created", "").startswith("202"),  # 近期创建
        }

    @classmethod
    def check_trustworthiness(cls, package_name: str) -> dict:
        """评估包的可信度"""
        info = cls.check_package_info(package_name)
        if not info["exists"]:
            return {"trusted": False, "reasons": ["包在 PyPI 上不存在"]}

        warnings = []
        trusted = True

        # 1. Typosquatting 检查
        typo_warning = TyposquattingDetector.check_typosquatting(package_name)
        if typo_warning:
            warnings.append(typo_warning)
            trusted = False

        # 2. 新包但下载量大的异常情况
        if info.get("is_new_package"):
            warnings.append(f"包创建时间较新: {info.get('created', 'unknown')}")

        # 3. 缺少项目主页
        if not info.get("home_page"):
            warnings.append("缺少项目主页")

        return {"trusted": trusted, "warnings": warnings}


# ===== pip 配置：只从官方源安装 =====
"""
# pip.conf（Windows: %APPDATA%/pip/pip.ini）
[global]

trusted-host = pypi.org
               files.pythonhosted.org
# 禁止额外的 index
# extra-index-url =   # 不设置！防止从不可信源安装

# 验证包签名（如可用）
require-hashes = true
"""

# ===== uv 配置 =====

"""
```
