# 数据安全与隐私合规

## 概述

数据安全与隐私合规是 RAG 项目的法律底线。中国《个人信息保护法》（PIPL）和《数据安全法》对个人信息的收集、存储、使用、传输提出了严格要求。本文件覆盖法规核心要求、数据分类分级、加密标准、数据脱敏策略和日志脱敏实践。

> RAG 系统涉及文档检索，若知识库中包含个人信息，必须实施严格的数据安全控制。

---

## 核心规则
### 🔴 必须遵守 (MUST)

1. **个人信息最小化收集**：只收集业务必需的个人信息
2. **脱敏入库**：包含个人信息的文档必须脱敏后才能写入 RAG 知识库
3. **传输加密**：所有 API 通信必须使用 TLS 1.2+
4. **存储加密**：数据库和文件系统必须启用静态加密（AES-256-GCM）
5. **日志脱敏**：禁止在日志中记录密码、Token、身份证号、手机号等
6. **数据分类分级**：明确定义数据等级并实施对应控制措施
7. **数据访问审计**：记录所有敏感数据的访问日志

### 🟡 强烈建议 (SHOULD)

1. 定期进行数据安全风险评估
2. 制定数据泄露应急预案
3. 实施数据生命周期管理（保留、清理、归档）

### 🟢 可选建议 (MAY)

1. 数据分级差异化管理策略
2. 同态加密技术（对极敏感数据的计算）

```python
from enum import IntEnum
from typing import Set


class DataLevel(IntEnum):
    """数据安全等级（数字越大，保护要求越高）"""

    CONFIDENTIAL = 2 # 敏感数据
    RESTRICTED = 3   # 核心数据（高敏感）

    PHONE = "phone"                # 手机号
    EMAIL = "email"                # 邮箱
    ADDRESS = "address"            # 住址
    # 企业数据

    INTERNAL_DOC = "internal_doc"  # 内部文档

DATA_CLASSIFICATION: dict[SensitiveDataType, DataLevel] = {
    # 核心数据（最高保护等级）
    SensitiveDataType.IDENTITY: DataLevel.RESTRICTED,
    SensitiveDataType.BIOMETRIC: DataLevel.RESTRICTED,
    SensitiveDataType.FINANCE: DataLevel.RESTRICTED,

}


def get_data_level(data_type: SensitiveDataType) -> DataLevel:
    """获取数据类型的安全等级"""
    return DATA_CLASSIFICATION.get(data_type, DataLevel.PUBLIC)


def get_required_controls(data_level: DataLevel) -> dict:
    """根据数据等级获取必须的安全控制措施"""
    controls = {

            "encryption_in_transit": True,

            "encryption_at_rest": True,

            "audit_logging": True,
            "masking_display": False,
        },
        DataLevel.CONFIDENTIAL: {
            "encryption_at_rest": True,     # AES-256-GCM
            "encryption_in_transit": True,  # TLS 1.3

            "encryption_at_rest": True,     # AES-256-GCM

            "access_control": True,         # 双人审批
            "audit_logging": True,          # 全量审计
```

### 存储加密：AES-256-GCM

```python
from cryptography.hazmat.primitives.ciphers.aead import AESGCM
import os
import base64


class DataEncryptor:
    """AES-256-GCM 数据加密器"""

    KEY_SIZE = 32  # 256 bits
    NONCE_SIZE = 12  # 96 bits (GCM 推荐)

    @classmethod
    def generate_key(cls) -> bytes:
        """生成 AES-256 密钥"""
        return AESGCM.generate_key(bit_length=256)

    @classmethod
    def encrypt(cls, plaintext: bytes, key: bytes) -> bytes:
        """
        加密数据。
        输出格式: nonce (12B) + ciphertext + tag (16B)
        """
        aesgcm = AESGCM(key)
        nonce = os.urandom(cls.NONCE_SIZE)
        ciphertext = aesgcm.encrypt(nonce, plaintext, None)
        # nonce 拼在前端，解密时需要
        return nonce + ciphertext

    @classmethod
    def decrypt(cls, encrypted: bytes, key: bytes) -> bytes:
        """解密数据"""
        aesgcm = AESGCM(key)
        nonce = encrypted[:cls.NONCE_SIZE]
        ciphertext = encrypted[cls.NONCE_SIZE:]
        return aesgcm.decrypt(nonce, ciphertext, None)

    @classmethod
    def encrypt_file(cls, input_path: str, output_path: str, key: bytes):
        """加密文件（分块处理大文件）"""
        aesgcm = AESGCM(key)
        nonce = os.urandom(cls.NONCE_SIZE)

        with open(input_path, "rb") as fin, open(output_path, "wb") as fout:
            # 写入 nonce
            fout.write(nonce)
            while True:
                chunk = fin.read(65536)  # 64KB 分块
                if not chunk:
                    break
                encrypted_chunk = aesgcm.encrypt(nonce, chunk, None)
                fout.write(encrypted_chunk)

    @classmethod
    def encrypt_sensitive_field(cls, value: str, key: bytes) -> str:
        """加密敏感字段（返回 base64 字符串）"""
        encrypted = cls.encrypt(value.encode("utf-8"), key)
        return base64.b64encode(encrypted).decode("utf-8")

    @classmethod
    def decrypt_sensitive_field(cls, encrypted_b64: str, key: bytes) -> str:
        """解密敏感字段"""
        encrypted = base64.b64decode(encrypted_b64)
        decrypted = cls.decrypt(encrypted, key)
        return decrypted.decode("utf-8")
```

## 6. 数据脱敏策略

```python
import re
from typing import Optional


class DataMasker:
    """数据脱敏工具"""

    @staticmethod
    def mask_id_card(id_card: str) -> str:
        """
        身份证号脱敏：保留前6位和后4位
        例如: 110101199001011234 → 110101********1234
        """
        if len(id_card) != 18:
            return "***"
        return id_card[:6] + "********" + id_card[-4:]

    @staticmethod
    def mask_phone(phone: str) -> str:
        """
        手机号脱敏：保留前3位和后4位
        例如: 13800138000 → 138****8000
        """
        if len(phone) != 11:
            return "***"
        return phone[:3] + "****" + phone[-4:]

    @staticmethod
    def mask_email(email: str) -> str:
        """
        邮箱脱敏
        例如: user@example.com → u***@example.com
        """
        parts = email.split("@")
        if len(parts) != 2:
            return "***@***"
        name, domain = parts
        if len(name) <= 1:
            masked_name = name + "***"
        else:
            masked_name = name[0] + "***" + name[-1]
        return f"{masked_name}@{domain}"

    @staticmethod
    def mask_name(name: str) -> str:
        """
        姓名脱敏
        例如: 张三 → 张**
        例如: 李小明 → 李**
        例如: 欧阳雪 → 欧**雪
        """
        if len(name) <= 1:
            return "*"
        elif len(name) == 2:
            return name[0] + "*"
        else:
            return name[0] + "**" + name[-1]

    @staticmethod
    def mask_bank_card(card_no: str) -> str:
        """
        银行卡号脱敏：保留前6位和后4位
        例如: 6222021234567890123 → 622202 ******* 0123
        """
        if len(card_no) < 10:
            return "****"
        return card_no[:6] + " ******* " + card_no[-4:]

    @staticmethod
    def mask_api_key(api_key: str) -> str:
        """API Key 脱敏"""
        if len(api_key) <= 8:
            return "****"
        return api_key[:4] + "****" + api_key[-4:]


# ===== 批量脱敏处理器 =====
class DocumentMasker:
    """文档脱敏处理器——用于 RAG 知识库入库前脱敏"""

    # 敏感信息的正则模式
    PATTERNS = {
        "id_card": re.compile(r"[1-9]\d{5}(?:19|20)\d{2}(?:0[1-9]|1[0-2])(?:0[1-9]|[12]\d|3[01])\d{3}[\dXx]"),
        "phone": re.compile(r"1[3-9]\d{9}"),
        "email": re.compile(r"\b[\w\.-]+@[\w\.-]+\.\w{2,4}\b"),
        "bank_card": re.compile(r"\b(?:62|60|58|56|55|54|53|52|51|50|49|48|47|46|45|44|43|42|41|40)\d{14,17}\b"),
    }

    @classmethod
    def mask_document(cls, content: str) -> str:
        """对文档中的敏感信息进行脱敏处理"""

        # 身份证号脱敏
        def replace_id_card(m: re.Match) -> str:
            return DataMasker.mask_id_card(m.group(0))

        # 手机号脱敏
        def replace_phone(m: re.Match) -> str:
            return DataMasker.mask_phone(m.group(0))

        # 邮箱脱敏
        def replace_email(m: re.Match) -> str:
            return DataMasker.mask_email(m.group(0))

        content = cls.PATTERNS["id_card"].sub(replace_id_card, content)
        content = cls.PATTERNS["phone"].sub(replace_phone, content)
        content = cls.PATTERNS["email"].sub(replace_email, content)
        content = cls.PATTERNS["bank_card"].sub(
            lambda m: DataMasker.mask_bank_card(m.group(0)),
            content,
        )

        return content


# ===== RAG 检索脱敏 =====
class RAGRetrievalMasker:
    """RAG 检索结果脱敏"""

    @classmethod
    def mask_retrieved_chunks(cls, chunks: list[dict]) -> list[dict]:
        """
        对检索到的文档块进行脱敏处理。
        在返回给用户之前执行。
        """
        masked_chunks = []
        for chunk in chunks:
            masked_chunk = chunk.copy()
            masked_chunk["content"] = DocumentMasker.mask_document(
                chunk["content"]
            )
            masked_chunks.append(masked_chunk)
        return masked_chunks
```

## 7. 日志脱敏

### 正确示例：日志脱敏实现

```python
import re
import json
import logging
from typing import Any
from datetime import datetime, timezone


# ===== 日志脱敏器 =====
class LogSanitizer:
    """日志脱敏处理器——确保敏感的字段不会出现在日志中"""

    # 敏感字段名（匹配字典 key）
    SENSITIVE_FIELDS = {
        "password", "passwd", "secret", "token", "api_key", "apikey",
        "access_token", "refresh_token", "jwt", "authorization",
        "auth_token", "session_id", "sessionid", "csrf_token",
        "credit_card", "card_number", "cvv", "cvc",
        "ssn", "id_card", "idcard", "phone", "mobile",
        "private_key", "secret_key", "ssh_key",
    }

    # 敏感值模式
    SENSITIVE_PATTERNS = [
        # API Key 格式
        (re.compile(r'(?i)(api[_-]?key|apikey|secret)[\s:=]+["\']?[\w-]{16,}'), r'\1=[REDACTED]'),
        # Token 格式
        (re.compile(r'(?i)(bearer\s+)[\w.-]{20,}'), r'\1[REDACTED]'),
        # 密码字段
        (re.compile(r'(?i)(password|passwd)[\s:=]+["\']?[^\s,;\'\"]{4,}'), r'\1=[REDACTED]'),
        # 身份证
        (re.compile(r'[1-9]\d{5}(?:19|20)\d{2}(?:0[1-9]|1[0-2])(?:0[1-9]|[12]\d|3[01])\d{3}[\dXx]'), '[REDACTED_ID]'),
        # 手机号
        (re.compile(r'1[3-9]\d[\d ]{8}\d'), '[REDACTED_PHONE]'),
    ]

    @classmethod
    def sanitize_log(cls, data: Any) -> Any:
        """
        递归脱敏日志数据。

        对字典的 key 进行检查：
        - 如果 key 在敏感字段列表中 → 值替换
        - 如果 value 是敏感数据 → 值替换
        """
        if isinstance(data, dict):
            sanitized = {}
            for key, value in data.items():
                # 检查 key 是否敏感
                if any(s in key.lower() for s in cls.SENSITIVE_FIELDS):
                    sanitized[key] = "[REDACTED]"
                else:
                    sanitized[key] = cls.sanitize_log(value)
            return sanitized

        elif isinstance(data, list):
            return [cls.sanitize_log(item) for item in data]

        elif isinstance(data, str):
            return cls._sanitize_string(data)

        else:
            return data

    @classmethod
    def _sanitize_string(cls, text: str) -> str:
        """对字符串中的敏感模式进行脱敏"""
        for pattern, replacement in cls.SENSITIVE_PATTERNS:
            text = pattern.sub(replacement, text)
        return text


# ===== 安全的日志记录器 =====
class SecurityAwareLogger:
    """安全感知的日志记录器——自动脱敏"""

    def __init__(self, name: str, level: int = logging.INFO):
        self.logger = logging.getLogger(name)
        self.logger.setLevel(level)

    def info(self, message: str, **kwargs):
        """记录 INFO 级别日志（自动脱敏）"""
        self._log(logging.INFO, message, **kwargs)

    def warning(self, message: str, **kwargs):
        """记录 WARNING 级别日志"""
        self._log(logging.WARNING, message, **kwargs)

    def error(self, message: str, **kwargs):
        """记录 ERROR 级别日志"""
        self._log(logging.ERROR, message, **kwargs)

    def _log(self, level: int, message: str, **kwargs):
        """内部日志方法——自动脱敏"""
        # 脱敏结构化的额外数据
        safe_kwargs = LogSanitizer.sanitize_log(kwargs)

        # 脱敏消息中的敏感信息
        safe_message = LogSanitizer._sanitize_string(message)

        self.logger.log(level, safe_message, extra=safe_kwargs)

        start = time.time()
        response = await call_next(request)

        # ✅ 正确：只记录非敏感信息
        self.logger.info(
            "API 请求",
            method=request.method,
            path=request.url.path,
            status_code=response.status_code,
            duration_ms=round((time.time() - start) * 1000, 2),
            client_ip=request.client.host if request.client else "unknown",
            # 🚫 不记录：request.headers.get("Authorization") — 包含 Token！
            # 🚫 不记录：request.query_params — 可能包含密码
        )

        return response


# ===== 实例化 =====
logger = SecurityAwareLogger("rag_app")

# ✅ 安全的使用
logger.info("用户查询完成", action="search", user_id="u1234")

# 自动脱敏字典中的敏感字段
```

## 8. RAG 专项数据安全

```python
class RAGDataSecurityManager:
    """RAG 系统数据安全管理器"""

    @staticmethod
    def preprocess_document_for_rag(
        content: str,
        metadata: dict,
    ) -> tuple[str, dict]:
        """
        文档入库前的数据安全处理。

        1. PII 检测与脱敏
        2. 数据等级标注
        3. 合规检查
        """
        # Step 1: 检测 PII
        masked_content = DocumentMasker.mask_document(content)

        # Step 2: 标注数据等级
        data_level = DataLevel.INTERNAL  # 从元数据推断

        # Step 3: 添加合规元数据
        safe_metadata = {
            **metadata,
            "data_level": data_level.name,
            "pii_redacted": content != masked_content,
            "encrypted_at": datetime.now(timezone.utc).isoformat(),
            "retention_days": 365,  # 数据保留期限
        }

        return masked_content, safe_metadata

    @staticmethod
    def should_allow_retrieval(
        user_level: DataLevel,
        doc_level: DataLevel,
    ) -> bool:
        """
        用户能否检索到特定级别的文档？
        规则：用户等级 >= 文档等级 才允许。
        """
        return user_level.value >= doc_level.value

    @staticmethod
    def sanitize_query_for_logging(query: str) -> str:
        """对用户查询进行脱敏（用于日志记录）"""
        return LogSanitizer._sanitize_string(query)

    @staticmethod
    def get_data_retention_policy(doc_level: DataLevel) -> dict:
        """获取数据保留策略"""
        policies = {
            DataLevel.PUBLIC: {"retention_days": 0},          # 不限
            DataLevel.INTERNAL: {"retention_days": 730},      # 2年
            DataLevel.CONFIDENTIAL: {"retention_days": 365},   # 1年
            DataLevel.RESTRICTED: {"retention_days": 90},     # 90天
        }
        return policies[doc_level]
```
